async function uploadResume() { const fileInput = document.getElementById('resume-file'); const file = fileInput.files[0]; const errorMessage = document.getElementById('error-message'); const result = document.getElementById('result'); errorMessage.style.display = 'none'; result.innerHTML = ''; if (!file) { errorMessage.textContent = 'Please select a resume file to upload.'; errorMessage.style.display = 'block'; return; } const reader = new FileReader(); reader.onload = async function(event) { const resumeText = event.target.result; try { const response = await fetch("https://your-vercel-url/api/claude", { // این URL همان API Vercel شما است method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ resume: resumeText }) }); const responseData = await response.json(); if (!response.ok) { throw new Error(`Error: ${response.status} - ${response.statusText}`); } result.innerHTML = `Improved Resume:
${responseData.messages[0].content}`; } catch (error) { console.error('Error occurred:', error); errorMessage.textContent = error.message; errorMessage.style.display = 'block'; } }; reader.readAsText(file); }
